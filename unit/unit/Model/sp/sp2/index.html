<script type="text/javascript">
    window.on_ahp_init = function () {
        function showVideo(domID, vid, url, auto) {
            auto = false;
            if (arguments.length >= 3) {
                if (typeof arguments[2] != 'string' || arguments[2] == 1) {
                    auto = arguments[2];
                }
            }
            var player = AHVP.newPlayer({
                vid: vid,
                container: domID,
                //指定是否自动播放，值为 1 或 true 时为自动播放，默认false 不自动播放 （此参数仅对PC端生效）
                autostart: auto
            });
            if (arguments.length >= 3) {
                if (typeof arguments[2] == 'string') {
                    player.addEventListener(AHVP.PLAYER_EVENT.VIDEO_START, function (e) {
                        jQuery.get(url);
                    });
                }
            }
        }

        //单个视频
        showVideo("playerContainer", "2F7313D234F3713D",true)
  }
</script>
