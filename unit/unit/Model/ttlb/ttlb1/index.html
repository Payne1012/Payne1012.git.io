<script type="text/javascript">

        if ($(window).width() < 1320) $(".rBox").addClass("rBox_");

        var d = $(".bg1 .toc"), time, i = 0, auto = true;
        $(function () {
            clearTimeout(time);
            time = setTimeout(function () { autoPlay(i) }, 5000)
        });
        function autoPlay(i) {
            i = i + 1 >= d.length ? 0 : i + 1;
            d.eq(i).css("z-index", 3).stop(true, true).fadeIn(900, function () {
                d.eq(i).siblings().css("z-index", 1).hide();
                d.eq(i).css("z-index", 2);
            });
            $(".doc a").eq(i).siblings().removeClass('docur');
            $(".doc a").eq(i).addClass('docur');
            clearTimeout(time);
            time = setTimeout(function () {
                if (auto) autoPlay(i);
            }, 5000)
        }
        $(".bg1").hover(function () {
            auto = false;
            clearTimeout(time);
        }, function () {
            auto = true;
            clearTimeout(time);
            time = setTimeout(function () { if (auto) autoPlay(i); }, 5000)
        });

        $(".test1").click(function () {
            $(".bg1 .toc").stop(true, true);
